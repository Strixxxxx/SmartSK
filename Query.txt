-- ============================================================================
-- MIGRATION SCRIPT: ADD HASH COLUMNS FOR DUPLICATION CHECKS
-- ============================================================================
-- Purpose: Add phoneHash to userInfo and preUserInfo tables.
--          Also ensures userInfo has emailHash and usernameHash for consistency.
-- ============================================================================

PRINT 'Starting hash column migration...';

-- Add phoneHash to userInfo
IF NOT EXISTS (
    SELECT 1 FROM sys.columns 
    WHERE object_id = OBJECT_ID('userInfo') 
    AND name = 'phoneHash'
)
BEGIN
    ALTER TABLE userInfo 
        ADD phoneHash VARCHAR(64) NULL;
    PRINT 'Added phoneHash column to userInfo';
END
ELSE
BEGIN
    PRINT 'phoneHash column already exists in userInfo';
END
GO

-- Add emailHash to userInfo (for consistency, if it was missed)
IF NOT EXISTS (
    SELECT 1 FROM sys.columns 
    WHERE object_id = OBJECT_ID('userInfo') 
    AND name = 'emailHash'
)
BEGIN
    ALTER TABLE userInfo 
        ADD emailHash VARCHAR(64) NULL;
    PRINT 'Added emailHash column to userInfo';
END
GO

-- Add usernameHash to userInfo (for consistency, if it was missed)
IF NOT EXISTS (
    SELECT 1 FROM sys.columns 
    WHERE object_id = OBJECT_ID('userInfo') 
    AND name = 'usernameHash'
)
BEGIN
    ALTER TABLE userInfo 
        ADD usernameHash VARCHAR(64) NULL;
    PRINT 'Added usernameHash column to userInfo';
END
GO

-- Add phoneHash to preUserInfo
IF NOT EXISTS (
    SELECT 1 FROM sys.columns 
    WHERE object_id = OBJECT_ID('preUserInfo') 
    AND name = 'phoneHash'
)
BEGIN
    ALTER TABLE preUserInfo 
        ADD phoneHash VARCHAR(64) NULL;
    PRINT 'Added phoneHash column to preUserInfo';
END
ELSE
BEGIN
    PRINT 'phoneHash column already exists in preUserInfo';
END
GO

-- Create indexes on new hash columns for faster lookups
IF NOT EXISTS (
    SELECT 1 FROM sys.indexes 
    WHERE name = 'IX_UserInfo_PhoneHash' 
    AND object_id = OBJECT_ID('userInfo')
)
BEGIN
    CREATE INDEX IX_UserInfo_PhoneHash ON userInfo(phoneHash) WHERE phoneHash IS NOT NULL;
    PRINT 'Created index on userInfo(phoneHash)';
END
GO

IF NOT EXISTS (
    SELECT 1 FROM sys.indexes 
    WHERE name = 'IX_PreUserInfo_PhoneHash' 
    AND object_id = OBJECT_ID('preUserInfo')
)
BEGIN
    CREATE INDEX IX_PreUserInfo_PhoneHash ON preUserInfo(phoneHash) WHERE phoneHash IS NOT NULL;
    PRINT 'Created index on preUserInfo(phoneHash)';
END
GO

PRINT 'Hash column migration script completed.';
PRINT 'NEXT STEP: Run the provided Node.js migration script to back-fill the hash values for existing data.';
