--Status Creation for the Project

CREATE PROCEDURE StatusNames
AS
BEGIN
    SELECT
        p.projectID,
        p.reference_number,
        p.title,
        p.description,
        p.submittedDate,
        p.reviewedBy,
        p.remarks,
        s.StatusName 
    FROM
        projects p
    JOIN
        StatusLookup s ON p.status = s.StatusID;
END;
GO

--Compilation of the Raw Data

CREATE PROCEDURE [Raw Data]
    @ppaFilter NVARCHAR(255) = NULL,
    @committeeFilter NVARCHAR(255) = NULL,
    @categoryFilter NVARCHAR(255) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- Get all distinct years from rawData table
    DECLARE @Years TABLE (year INT);
    INSERT INTO @Years (year)
    SELECT DISTINCT year
    FROM rawData
    WHERE year IS NOT NULL
    ORDER BY year;

    -- Check if any years were found
    IF NOT EXISTS (SELECT 1 FROM @Years)
    BEGIN
        SELECT
            CAST(NULL AS NVARCHAR(255)) AS ppa,
            CAST(NULL AS NVARCHAR(255)) AS category,
            CAST(NULL AS NVARCHAR(255)) AS committee,
            CAST(NULL AS INT) AS dummy_target, -- Placeholder columns for an empty set
            CAST(NULL AS INT) AS dummy_budget
        WHERE 1 = 0;
        RETURN;
    END;

    -- Build the dynamic part of the SQL query for year-based columns
    DECLARE @DynamicPivotColumns NVARCHAR(MAX) = N'';
    DECLARE @CurrentYear INT;

    DECLARE years_cursor CURSOR FOR SELECT year FROM @Years;
    OPEN years_cursor;
    FETCH NEXT FROM years_cursor INTO @CurrentYear;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        -- **FIX IS HERE:** Changed 'T' to 'rd' (the alias for rawData in the main query)
        SET @DynamicPivotColumns = @DynamicPivotColumns +
            N', MAX(CASE WHEN rd.year = ' + CAST(@CurrentYear AS NVARCHAR(4)) + N' THEN rd.target END) AS [' + CAST(@CurrentYear AS NVARCHAR(4)) + N'_target]' +
            N', MAX(CASE WHEN rd.year = ' + CAST(@CurrentYear AS NVARCHAR(4)) + N' THEN rd.budget END) AS [' + CAST(@CurrentYear AS NVARCHAR(4)) + N'_budget]';

        FETCH NEXT FROM years_cursor INTO @CurrentYear;
    END;
    CLOSE years_cursor;
    DEALLOCATE years_cursor;

    -- Build the main query with filters and dynamic columns
    DECLARE @SqlQuery NVARCHAR(MAX);
    SET @SqlQuery = N'
        SELECT
            rdd.ppa,
            rdd.category,
            rdd.committee' + @DynamicPivotColumns + N'
        FROM rawDataDetails rdd
        LEFT JOIN rawData rd ON rdd.rddID = rd.rddID
        WHERE 1=1'; -- Placeholder to easily append WHERE clauses

    -- Add conditional WHERE clauses for filtering (using parameterized execution for security)
    IF @ppaFilter IS NOT NULL AND LEN(@ppaFilter) > 0
        SET @SqlQuery = @SqlQuery + N' AND rdd.ppa LIKE @ppaPattern';

    IF @committeeFilter IS NOT NULL AND LEN(@committeeFilter) > 0
        SET @SqlQuery = @SqlQuery + N' AND rdd.committee = @committeeFilter';

    IF @categoryFilter IS NOT NULL AND LEN(@categoryFilter) > 0
        SET @SqlQuery = @SqlQuery + N' AND rdd.category = @categoryFilter';

    SET @SqlQuery = @SqlQuery + N'
        GROUP BY
            rdd.ppa,
            rdd.category,
            rdd.committee
        ORDER BY
            rdd.ppa, rdd.category, rdd.committee;';

    -- Prepare and execute the dynamic SQL
    DECLARE @Params NVARCHAR(MAX) = N'@ppaPattern NVARCHAR(255), @committeeFilter NVARCHAR(255), @categoryFilter NVARCHAR(255)';
    DECLARE @PpaPatternValue NVARCHAR(255) = '%' + @ppaFilter + '%';

    EXEC sp_executesql @SqlQuery, @Params,
        @ppaPattern = @PpaPatternValue,
        @committeeFilter = @committeeFilter,
        @categoryFilter = @categoryFilter;
END;

--Project Archiving

CREATE PROCEDURE prjArchived
    -- Input parameter: The ID of the project to archive
    @projectID INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Start a transaction for the move operation.
    BEGIN TRANSACTION;

    BEGIN TRY
        -- Check if the project exists before attempting to move it.
        IF EXISTS (SELECT 1 FROM dbo.projects WHERE projectID = @projectID)
        BEGIN
            -- Enable IDENTITY_INSERT to preserve the original projectID.
            SET IDENTITY_INSERT dbo.projectsARC ON;

            -- Step 1: Copy the project data to the archive table.
            INSERT INTO dbo.projectsARC (
                projectID, reference_number, title, description, userID, status,
                submittedDate, file_path, file_name, remarks, reviewedBy
            )
            SELECT
                projectID, reference_number, title, description, userID, status,
                submittedDate, file_path, file_name, remarks, reviewedBy
            FROM
                dbo.projects
            WHERE
                projectID = @projectID;

            -- Disable IDENTITY_INSERT.
            SET IDENTITY_INSERT dbo.projectsARC OFF;

            -- Step 2: Delete the project from the main table.
            DELETE FROM dbo.projects
            WHERE projectID = @projectID;

            -- If all goes well, commit the changes.
            COMMIT TRANSACTION;
            PRINT 'Project with ID ' + CAST(@projectID AS VARCHAR) + ' was successfully archived.';
        END
        ELSE
        BEGIN
             -- If project does not exist, rollback to do nothing and print a message.
            ROLLBACK TRANSACTION;
            PRINT 'Error: Project with ID ' + CAST(@projectID AS VARCHAR) + ' not found. No action taken.';
        END
    END TRY
    BEGIN CATCH
        -- If an error occurs, undo all changes.
        ROLLBACK TRANSACTION;
        PRINT 'An error occurred. Transaction rolled back.';
        THROW;
    END CATCH
END
GO

--Projects return

CREATE PROCEDURE prjReturn
    -- Input parameter: The ID of the project to return from the archive
    @projectID INT
AS
BEGIN
    SET NOCOUNT ON;

    BEGIN TRANSACTION;

    BEGIN TRY
        -- PRE-FLIGHT CHECKS --
        -- 1. Check if the project is in the archive.
        IF NOT EXISTS (SELECT 1 FROM dbo.projectsARC WHERE projectID = @projectID)
        BEGIN
            PRINT 'Error: Project with ID ' + CAST(@projectID AS VARCHAR) + ' not found in the archive. No action taken.';
            ROLLBACK TRANSACTION;
            RETURN;
        END

        -- 2. Check if the project ID already exists in the main projects table.
        IF EXISTS (SELECT 1 FROM dbo.projects WHERE projectID = @projectID)
        BEGIN
            PRINT 'Error: Project with ID ' + CAST(@projectID AS VARCHAR) + ' already exists in the main projects table. Cannot restore.';
            ROLLBACK TRANSACTION;
            RETURN;
        END

        -- Enable IDENTITY_INSERT on the destination table.
        SET IDENTITY_INSERT dbo.projects ON;

        -- Step 1: Copy the project data from the archive back to the main table.
        INSERT INTO dbo.projects (
            projectID, reference_number, title, description, userID, status,
            submittedDate, file_path, file_name, remarks, reviewedBy
        )
        SELECT
            projectID, reference_number, title, description, userID, status,
            submittedDate, file_path, file_name, remarks, reviewedBy
        FROM
            dbo.projectsARC
        WHERE
            projectID = @projectID;

        -- Disable IDENTITY_INSERT.
        SET IDENTITY_INSERT dbo.projects OFF;

        -- Step 2: Delete the project from the archive table.
        DELETE FROM dbo.projectsARC
        WHERE projectID = @projectID;

        -- Commit the transaction if everything succeeded.
        COMMIT TRANSACTION;
        PRINT 'Project with ID ' + CAST(@projectID AS VARCHAR) + ' was successfully returned from the archive.';

    END TRY
    BEGIN CATCH
        -- If any error happens, cancel the transaction.
        ROLLBACK TRANSACTION;
        PRINT 'An error occurred. Transaction rolled back.';
        THROW;
    END CATCH
END
GO
