--Initialization of the Project Batch

DROP PROCEDURE IF EXISTS sp_InitializeProjectBatch;
GO

CREATE PROCEDURE sp_InitializeProjectBatch
    @barangayID INT,
    @projType NVARCHAR(10),
    @projName NVARCHAR(255),
    @targetYear NVARCHAR(20),
    @budget MONEY,
    @userID INT,
    @governance_pct DECIMAL(5,2) = 0,
    @active_citizenship_pct DECIMAL(5,2) = 0,
    @economic_empowerment_pct DECIMAL(5,2) = 0,
    @global_mobility_pct DECIMAL(5,2) = 0,
    @agriculture_pct DECIMAL(5,2) = 0,
    @environment_pct DECIMAL(5,2) = 0,
    @PBS_pct DECIMAL(5,2) = 0,
    @SIE_pct DECIMAL(5,2) = 0,
    @education_pct DECIMAL(5,2) = 0,
    @health_pct DECIMAL(5,2) = 0
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        DECLARE @total_pct DECIMAL(5,2) = 
            @governance_pct + @active_citizenship_pct + @economic_empowerment_pct + 
            @global_mobility_pct + @agriculture_pct + @environment_pct + 
            @PBS_pct + @SIE_pct + @education_pct + @health_pct;

        -- Insert into header with centralized barangayID, projName, and percentages
        INSERT INTO projectBatch (
            barangayID, projType, projName, targetYear, budget,
            governance_pct, active_citizenship_pct, economic_empowerment_pct,
            global_mobility_pct, agriculture_pct, environment_pct,
            PBS_pct, SIE_pct, education_pct, health_pct, total_pct
        )
        VALUES (
            @barangayID, @projType, @projName, @targetYear, @budget,
            @governance_pct, @active_citizenship_pct, @economic_empowerment_pct,
            @global_mobility_pct, @agriculture_pct, @environment_pct,
            @PBS_pct, @SIE_pct, @education_pct, @health_pct, @total_pct
        );
        
        DECLARE @newBatchID INT = SCOPE_IDENTITY();
        
        -- Start Project Tracker at Step 1 (Template Setup)
        INSERT INTO projectTracker (batchID, statusID, updatedBy)
        VALUES (@newBatchID, 1, @userID);
        
        SELECT @newBatchID AS batchID;
        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
GO


--Fetching Data to be Displayed on Barangay Dashboard

DROP PROCEDURE IF EXISTS sp_GetBarangayDashboard;
GO

CREATE PROCEDURE sp_GetBarangayDashboard
    @barangayID INT
AS
BEGIN
    SET NOCOUNT ON;
    -- Returns unique batches with their latest status for a specific barangay
    SELECT 
        pb.*,
        sl.StatusName,
        sl.StatusID as currentStatusID
    FROM projectBatch pb
    CROSS APPLY (
        SELECT TOP 1 pt.statusID 
        FROM projectTracker pt 
        WHERE pt.batchID = pb.batchID 
        ORDER BY pt.updatedAt DESC
    ) latestStatus
    JOIN StatusLookup sl ON latestStatus.statusID = sl.StatusID
    WHERE pb.barangayID = @barangayID
    ORDER BY pb.createdAt DESC;
END;
GO

--Transfer of the Account from the preUserInfo to UserInfo Table

CREATE PROCEDURE [sp_ApprovePendingUser]
    @userID INT
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    BEGIN TRY
        -- Ensure the user is not already in the main userInfo table
        IF NOT EXISTS (SELECT 1 FROM userInfo WHERE userID = @userID)
        BEGIN
            -- Use SET IDENTITY_INSERT to keep the same userID
            SET IDENTITY_INSERT userInfo ON;

            -- Insert the approved user's data into the main table
            INSERT INTO userInfo (
                userID, username, passKey, fullName, position, barangay, 
                emailAddress, phoneNumber, isDefaultPassword, isArchived, 
                emailHash, usernameHash, phoneHash, encryptionKeyVersion
            )
            SELECT
                p.userID, p.username, p.passKey, p.fullName, p.position, p.barangay,
                p.emailAddress, p.phoneNumber, 0, p.isArchived, -- Set isDefaultPassword to 0 as per requirements
                p.emailHash, p.usernameHash, p.phoneHash, encryptionKeyVersion
            FROM preUserInfo p
            WHERE p.userID = @userID;

            SET IDENTITY_INSERT userInfo OFF;

            -- Update the status in the extension table to 'approved'
            UPDATE preUserInfoEx
            SET status = 'approved'
            WHERE userID = @userID;
        END
        ELSE
        BEGIN
            -- If user already exists, just ensure their status is marked as approved
            UPDATE preUserInfoEx
            SET status = 'approved', rejectionReason = 'User already existed in main table but was re-approved.'
            WHERE userID = @userID;
        END

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        -- Re-throw the error to be caught by the calling application
        THROW;
    END CATCH
END
GO

--Checking for Project Deadlines

DROP PROCEDURE IF EXISTS sp_CheckProjectDeadlines;
GO

CREATE PROCEDURE sp_CheckProjectDeadlines
AS
BEGIN
SET NOCOUNT ON;

-- CTE: Get the latest status and date for each batch
WITH LatestStatus AS (
    SELECT
        pt.batchID,
        pt.statusID,
        pt.updatedAt,
        DATEDIFF(DAY, pt.updatedAt, GETDATE()) AS daysStuck
    FROM projectTracker pt
    INNER JOIN (
        SELECT batchID, MAX(updatedAt) AS maxDate
        FROM projectTracker
        GROUP BY batchID
    ) latest ON pt.batchID = latest.batchID AND pt.updatedAt = latest.maxDate
)
INSERT INTO projectNotifications (batchID, barangayID, notifType, message)
SELECT
    pb.batchID,
    pb.barangayID,
    CASE
        WHEN ls.daysStuck >= 30 THEN 'URGENT'
        ELSE 'DEADLINE_WARNING'
    END,
    CASE
        WHEN ls.daysStuck >= 30 THEN
            'URGENT: Project "' + pb.projName + '" (' + pb.projType + ') has been in "' + sl.StatusName + '" for ' + CAST(ls.daysStuck AS NVARCHAR) + ' days. Immediate action is required.'
        ELSE
            'REMINDER: Project "' + pb.projName + '" (' + pb.projType + ') has been in "' + sl.StatusName + '" for ' + CAST(ls.daysStuck AS NVARCHAR) + ' days. Please review and advance the status.'
    END
FROM LatestStatus ls
JOIN projectBatch  pb ON ls.batchID  = pb.batchID
JOIN StatusLookup  sl ON ls.statusID = sl.StatusID
 WHERE
    pb.isArchived = 0
    AND ls.statusID NOT IN (8, 9)  -- Skip completed/closed projects
    AND ls.daysStuck >= 15          -- Only flag if 15+ days stale
    -- Avoid inserting a duplicate notification for same batch on same day
    AND NOT EXISTS (
        SELECT 1 FROM projectNotifications pn
        WHERE pn.batchID = pb.batchID
            AND CAST(pn.createdAt AS DATE) = CAST(GETDATE() AS DATE)
    );
END;
GO

--Manually Approved Account Application 

CREATE PROCEDURE [sp_ManuallyApprovePendingUser]
    @userID INT,
    @adminReport NVARCHAR(MAX),
    @adminUsername NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    DECLARE @attachmentPath NVARCHAR(MAX);
    DECLARE @attachmentPathBack NVARCHAR(MAX);

    BEGIN TRY
        SELECT 
            @attachmentPath = attachmentPath, 
            @attachmentPathBack = attachmentPathBack
        FROM preUserInfoEx 
        WHERE userID = @userID;

        IF NOT EXISTS (SELECT 1 FROM userInfo WHERE userID = @userID)
        BEGIN
            SET IDENTITY_INSERT userInfo ON;
            INSERT INTO userInfo (
                userID, username, passKey, fullName, position, barangay, 
                emailAddress, phoneNumber, isDefaultPassword, isArchived, 
                emailHash, usernameHash, phoneHash, encryptionKeyVersion
            )
            SELECT
                p.userID, p.username, p.passKey, p.fullName, p.position, p.barangay,
                p.emailAddress, p.phoneNumber, 0, p.isArchived,
                p.emailHash, p.usernameHash, p.phoneHash, p.encryptionKeyVersion
            FROM preUserInfo p
            WHERE p.userID = @userID;
            SET IDENTITY_INSERT userInfo OFF;
        END

        UPDATE preUserInfoEx
        SET status = 'approved', rejectionReason = NULL
        WHERE userID = @userID;

        -- Log the manual approval with isApprove = 1
        INSERT INTO registrationAuditManual (userID, verificationReport, attachmentPath, attachmentPathBack, processedBy, isApprove)
        VALUES (@userID, @adminReport, @attachmentPath, @attachmentPathBack, @adminUsername, 1);

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END
GO

--Manually Rejected Account Application

CREATE PROCEDURE [sp_ManuallyRejectApprovedUser]
    @userID INT,
    @adminReport NVARCHAR(MAX),
    @adminUsername NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    DECLARE @attachmentPath NVARCHAR(MAX);
    DECLARE @attachmentPathBack NVARCHAR(MAX);

    BEGIN TRY
        SELECT 
            @attachmentPath = attachmentPath, 
            @attachmentPathBack = attachmentPathBack
        FROM preUserInfoEx 
        WHERE userID = @userID;

        -- Log the manual rejection with isApprove = 0
        INSERT INTO registrationAuditManual (userID, verificationReport, attachmentPath, attachmentPathBack, processedBy, isApprove)
        VALUES (@userID, @adminReport, @attachmentPath, @attachmentPathBack, @adminUsername, 0);

        UPDATE preUserInfoEx
        SET status = 'rejected', rejectionReason = @adminReport
        WHERE userID = @userID;

        IF EXISTS (SELECT 1 FROM userInfo WHERE userID = @userID)
        BEGIN
            DELETE FROM otpStore WHERE userID = @userID;
            DELETE FROM sessions WHERE userID = @userID;
            DELETE FROM [audit trail] WHERE userID = @userID;
            DELETE FROM postComment WHERE userID = @userID;
            DELETE FROM posts WHERE userID = @userID;
            DELETE FROM projects WHERE userID = @userID;
            DELETE FROM BackupJobs WHERE UserID = @userID;
            DELETE FROM PostUploadJobs WHERE UserID = @userID;
            DELETE FROM rawDataUpTrack WHERE uploadedByUserID = @userID;
            DELETE FROM userInfo WHERE userID = @userID;
        END

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END
GO
