name: Build and deploy Node.js backend app to Azure WebApp

on:
  push:
    branches:
      - main
    paths:
      - "backend/**"

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: "22.x"

      - name: Install backend dependencies
        working-directory: ./backend
        run: npm ci

      # ✅ No need to copy node_modules into itself
      # node_modules is already inside backend after npm ci

      - name: Deploy to Azure WebApp
        uses: azure/webapps-deploy@v2
        with:
          app-name: "smartSK-backend" # your App Service name
          slot-name: "Production"
          package: ./backend
