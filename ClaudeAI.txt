typescriptimport React, { useState, useEffect } from 'react';
import axios from 'axios'; // Use regular axios, not axiosInstance
import { useWebSocket } from '../../context/WebSocketContext';
import Maintenance from './Maintenance';

interface MaintenanceHandlerProps {
    children: React.ReactNode;
}

const MaintenanceHandler: React.FC<MaintenanceHandlerProps> = ({ children }) => {
    const [maintenanceMode, setMaintenanceMode] = useState(false);
    const [checkingMaintenance, setCheckingMaintenance] = useState(true);
    const { maintenanceMessage } = useWebSocket();

    useEffect(() => {
        const checkMaintenanceStatus = async () => {
            try {
                // Use direct axios instead of axiosInstance to bypass auth checks
                const response = await axios.get(`${process.env.REACT_APP_API_URL || ''}/api/maintenance-status`);
                setMaintenanceMode(response.data.maintenance);
            } catch (error) {
                console.error('Failed to check maintenance status:', error);
                // On error, assume no maintenance to allow normal flow
                setMaintenanceMode(false);
            } finally {
                setCheckingMaintenance(false);
            }
        };

        checkMaintenanceStatus();
        const interval = setInterval(checkMaintenanceStatus, 30000); // Poll every 30 seconds
        return () => clearInterval(interval);
    }, []);

    useEffect(() => {
        if (maintenanceMessage?.type === 'maintenance_ended') {
            setMaintenanceMode(false);
            // Reload to clear state and force re-authentication
            window.location.reload();
        }
        if (maintenanceMessage?.type === 'maintenance_starting') {
            setMaintenanceMode(true);
        }
    }, [maintenanceMessage]);

    // Show loading while checking (optional - you could show nothing)
    if (checkingMaintenance) {
        return <div>Loading...</div>;
    }

    // PRIORITY: Show maintenance page BEFORE any other content
    if (maintenanceMode) {
        return <Maintenance />;
    }

    return <>{children}</>;
};

export default MaintenanceHandler;
